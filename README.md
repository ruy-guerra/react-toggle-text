## Install

```bash
npm install --save react-toggle-text
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-toggle-text'
import 'react-toggle-text/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
